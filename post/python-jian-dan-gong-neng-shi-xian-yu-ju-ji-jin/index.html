df['word'].cat.set_categories(word_order, inplace=True)
df.sort_values(by=["word","num"],inplace=True)

column_order=['cd','ac','bd','da']
df = df[column_order]

df.drop_duplicates(['colname1','colname2','colname3'], 'last')
